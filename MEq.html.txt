<script>
function calculateME() {
  // conversion factors: ME = (dose / max dose) * 100
  const midodrine = (parseFloat(document.getElementById('midodrine').value) || 0) / 30 * 100;
  const fludro = (parseFloat(document.getElementById('fludro').value) || 0) / 0.3 * 100;
  const droxidopa = (parseFloat(document.getElementById('droxidopa').value) || 0) / 1800 * 100;
  const pyridostigmine = (parseFloat(document.getElementById('pyridostigmine').value) || 0) / 180 * 100;

  const total = midodrine + fludro + droxidopa + pyridostigmine;
  document.getElementById('totalME').innerText = total.toFixed(1);
}
</script>
